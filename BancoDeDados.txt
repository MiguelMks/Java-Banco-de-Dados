CREATE TABLE tb_produtos(
id int primary key auto_increment,
descricao varchar(50),
preco decimal (6,2),
estoque int
);
CREATE TABLE tb_clientes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    cpf VARCHAR(100),
    nome VARCHAR(100),
    email VARCHAR(100),
    rua VARCHAR(100),
    numero INT,
    bairro VARCHAR(100),
    cep VARCHAR(80),
    cidade VARCHAR(50),
    estado VARCHAR(100)
);
CREATE TABLE tb_pedidos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    tb_cliente_id INT NOT NULL,
    data_pedido DATETIME DEFAULT CURRENT_TIMESTAMP,
    status VARCHAR(50) NOT NULL,
    valor_total DOUBLE NOT NULL,
    FOREIGN KEY (tb_cliente_id) REFERENCES tb_clientes(id)
);

CREATE TABLE tb_item_pedido (
    id INT AUTO_INCREMENT PRIMARY KEY,
    tb_pedidos_id INT NOT NULL,
    tb_produtos_id INT NOT NULL,
    quantidade INT NOT NULL,
    preco_unitario DECIMAL(10,2) NOT NULL,
    FOREIGN KEY (tb_pedidos_id) REFERENCES tb_pedidos(id),
    FOREIGN KEY (tb_produtos_id) REFERENCES tb_produtos(id)
);
